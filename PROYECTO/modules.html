Modules export and import

Características

Antes de ES6, se usaba librerías como Browserify para crear módulos en el lado del cliente, y require con Node.js. Con ES6, podemos usarlos directamente.

Exportar módulos
              

module.exports = 1;
module.exports = { foo: 'bar' };
module.exports = ['foo', 'bar'];
module.exports = function bar () {};

Con ES6 podemos exportar con nombre. 

              

export let name = 'David';
export let age  = 25;

También se puede exportar una lista de objetos. 

              

function sumTwo(a, b) {
    return a + b;
}

function sumThree(a, b, c) {
    return a + b + c;
}

export { sumTwo, sumThree };

También se pueden exportar funciones, objetos y valores con la palabra export. 

              

export function sumTwo(a, b) {
    return a + b;
}

export function sumThree(a, b, c) {
    return a + b + c;
}

También se pueden exportar bindings por defecto. 

              

function sumTwo(a, b) {
    return a + b;
}

function sumThree(a, b, c) {
    return a + b + c;
}

let api = {
    sumTwo,
    sumThree
};

export default api;

/* Which is the same as
 * export { api as default };
 */
Importar módulos

Se puede importar un archivo entero. 

              

import 'underscore';

Como en Python, tenemos importaciones con nombre. 

              

import {
    sumTwo as addTwoNumbers,
    sumThree as sumThreeNumbers
} from 'math/addition';

También podemos importar todas las cosas. 

              

import * as util from 'math/addition';

También podemos importar una lista de valores del módulo. 

              

import * as additionUtil from 'math/addition';
const { sumTwo, sumThree } = additionUtil;

También podemos importar bindings por defecto. 

              

import api from 'math/addition';
// Same as: import { default as api } from 'math/addition';

Es posible combinar exportaciones e importaciones por defecto de ser necesario. 

              

// foos.js
export { foo as default, foo1, foo2 };

import foo, { foo1, foo2 } from 'foos';

Cuando importamos un módulo exportado con commonjs podemos usar: 

              

import React from 'react';
const { Component, PropTypes } = React;

import React, { Component, PropTypes } from 'react';
