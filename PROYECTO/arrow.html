Arrow Functions

Características

Usualmente tenemos funciones anidadas donde quisieramos conservar el contexto de this de su alcance léxico. Por ejemplo: 

              

function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    return arr.map(function (character) {
        return this.name + character; // Cannot read property 'name' of undefined
    });
};

Una solución común a esto es guardar el contexto de this en una variable. 

              

function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    var that = this; // Store the context of this
    return arr.map(function (character) {
        return that.name + character;
    });
};

También podemos pasar como tal el contexto de this. 

              

function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    return arr.map(function (character) {
        return this.name + character;
    }, this);
};

También podemos enlazar el contexto. 

              

function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    return arr.map(function (character) {
        return this.name + character;
    }.bind(this));
};

Con las funciones flecha el contexto de this no es ocultado y podemos escribir las funciones así: 

              

function Person(name) {
    this.name = name;
}

Person.prototype.prefixName = function (arr) {
    return arr.map(character => this.name + character);
};

Las funciones flecha son más concisas cuando se usan en expresiones de funciones que simplemente retornan un valor. 

              

var squares = arr.map(function (x) { return x * x }); // Function Expression
              

            
              

const arr = [1, 2, 3, 4, 5];
const squares = arr.map(x => x * x); // Arrow Function for terser implementation
